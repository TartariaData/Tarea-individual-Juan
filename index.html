import matplotlib.pyplot as plt

def plot_map(data, column, title, cmap='viridis'):
    f, ax = plt.subplots(1, figsize=(10, 10))
    data.plot(column=column, cmap=cmap, categorical=True, edgecolor='white', linewidth=0.5, alpha=1, legend=True, legend_kwds={'loc': "lower left"}, ax=ax)
    ax.set_axis_off()
    ax.set_title(title)
    plt.show()
